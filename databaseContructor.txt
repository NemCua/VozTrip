-- Quan tri nguoi dung
CREATE TABLE users (
    user_id      TEXT PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
    username     VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    role         VARCHAR(20)  NOT NULL,   -- 'admin' | 'seller'
    full_name    VARCHAR(200),
    email        VARCHAR(200),
    is_active    BOOLEAN DEFAULT TRUE,
    created_at   TIMESTAMPTZ DEFAULT NOW()
);

-- Thong tin rieng cua seller
CREATE TABLE sellers (
    seller_id    TEXT PRIMARY KEY REFERENCES users(user_id),
    shop_name    VARCHAR(200) NOT NULL,
    shop_logo    VARCHAR(500),
    contact_phone VARCHAR(20),
    description  TEXT,
    approved_at  TIMESTAMPTZ,
    approved_by  TEXT REFERENCES users(user_id)
);

-- Khu vuc / Khu trung bay
CREATE TABLE zones (
    zone_id     TEXT PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
    zone_name   VARCHAR(100) NOT NULL,
    description TEXT
);

-- Diem tham quan / Gian hang
CREATE TABLE pois (
    poi_id         TEXT PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
    seller_id      TEXT NOT NULL REFERENCES sellers(seller_id),
    zone_id        TEXT REFERENCES zones(zone_id),
    poi_name       VARCHAR(200) NOT NULL,
    latitude       DOUBLE PRECISION NOT NULL,
    longitude      DOUBLE PRECISION NOT NULL,
    trigger_radius DOUBLE PRECISION DEFAULT 10.0,  -- met
    is_active      BOOLEAN DEFAULT TRUE,
    created_at     TIMESTAMPTZ DEFAULT NOW()
);

-- Ngon ngu ho tro
CREATE TABLE languages (
    language_id   TEXT PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
    language_code VARCHAR(10) UNIQUE NOT NULL,  -- 'en', 'zh', 'ko', 'ja'
    language_name VARCHAR(100),
    is_active     BOOLEAN DEFAULT TRUE
);

-- Noi dung + audio theo ngon ngu (gop PoiContents va PoiAudios)
CREATE TABLE poi_localizations (
    localization_id TEXT PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
    poi_id          TEXT NOT NULL REFERENCES pois(poi_id) ON DELETE CASCADE,
    language_id     TEXT NOT NULL REFERENCES languages(language_id),
    title           VARCHAR(200),
    description     TEXT,
    audio_url       VARCHAR(500),   -- Cloudinary URL
    audio_public_id VARCHAR(200),   -- Cloudinary public_id de xoa
    audio_duration  INTEGER,        -- giay
    UNIQUE (poi_id, language_id)
);

-- Anh va video cua POI (luu ca public_id cua Cloudinary de xoa duoc)
CREATE TABLE poi_media (
    media_id   TEXT PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
    poi_id     TEXT NOT NULL REFERENCES pois(poi_id) ON DELETE CASCADE,
    media_type VARCHAR(20) NOT NULL,   -- 'image' | 'video'
    media_url  VARCHAR(500) NOT NULL,  -- Cloudinary URL
    public_id  VARCHAR(200) NOT NULL,  -- Cloudinary public_id de xoa
    sort_order INTEGER DEFAULT 0
);

-- Cau hoi thuong gap
CREATE TABLE questions (
    question_id   TEXT PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
    poi_id        TEXT NOT NULL REFERENCES pois(poi_id) ON DELETE CASCADE,
    language_id   TEXT NOT NULL REFERENCES languages(language_id),
    question_text VARCHAR(500) NOT NULL,
    sort_order    INTEGER DEFAULT 0,
    is_active     BOOLEAN DEFAULT TRUE
);

-- Cau tra loi (giu them poi_id de query nhanh)
CREATE TABLE answers (
    answer_id   TEXT PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
    question_id TEXT NOT NULL REFERENCES questions(question_id) ON DELETE CASCADE,
    poi_id      TEXT NOT NULL REFERENCES pois(poi_id),
    language_id TEXT NOT NULL REFERENCES languages(language_id),
    answer_text TEXT NOT NULL,
    audio_url   VARCHAR(500),   -- Cloudinary URL
    audio_public_id VARCHAR(200) -- Cloudinary public_id de xoa
);

-- Phien khach (khong can dang nhap)
CREATE TABLE guest_sessions (
    session_id  TEXT PRIMARY KEY,   -- UUID tu app sinh ra
    language_id TEXT REFERENCES languages(language_id),
    started_at  TIMESTAMPTZ DEFAULT NOW()
);

-- Lich su tham quan
CREATE TABLE visit_logs (
    log_id       TEXT PRIMARY KEY DEFAULT gen_random_uuid()::TEXT,
    session_id   TEXT REFERENCES guest_sessions(session_id),
    poi_id       TEXT REFERENCES pois(poi_id),
    triggered_at TIMESTAMPTZ DEFAULT NOW()
);
